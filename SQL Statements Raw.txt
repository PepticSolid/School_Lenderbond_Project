CREATE TABLE accountHolder(user_id INTEGER PRIMARY KEY,username TEXT NOT NULL, pass TEXT NOT NULL);

CREATE TABLE standardAccount(standardAccount_id INTEGER PRIMARY KEY,user_id INTEGER NOT NULL,FOREIGN KEY (user_id) REFERENCES accountHolder(user_id)); 
CREATE TABLE businessAccount(businessAccount_id INTEGER PRIMARY KEY,user_id INTEGER NOT NULL,FOREIGN KEY (user_id) REFERENCES accountHolder(user_id), business_name TEXT NOT NULL);

CREATE TABLE messageBoard(MBthred_id INTEGER PRIMARY KEY, subjectTitle TEXT NOT NULL, dateCreated DATE);

CREATE TABLE Bond(bond_id INTEGER PRIMARY KEY,currency INTEGER NOT NULL,bond_name TEXT NOT NULL, MBthred_id INTEGER NOT NULL, FOREIGN KEY (MBthred_id) REFERENCES messageBoard(MBthred_id));

CREATE TABLE Loan(loan_id INTEGER PRIMARY KEY,init_currency INTEGER NOT NULL,interest_rate INTEGER NOT NULL,loan_name TEXT NOT NULL, MBthred_id INTEGER NOT NULL,FOREIGN KEY (MBthred_id) REFERENCES messageBoard(MBthred_id)); 

CREATE TABLE loansHeld(accountNumber INTEGER PRIMARY KEY, user_id INTEGER NOT NULL,FOREIGN KEY (user_id) REFERENCES accountHolder(user_id),loan_id INTEGER NOT NULL,FOREIGN KEY (loan_id) REFERENCES Loan(loan_id));
CREATE TABLE bondsHeld(accountNumber INTEGER PRIMARY KEY, user_id INTEGER NOT NULL,FOREIGN KEY (user_id) REFERENCES accountHolder(user_id),bond_id INTEGER NOT NULL,FOREIGN KEY (bond_id) REFERENCES Bond(bond_id));

CREATE TABLE MBmessages(MessageNumber INTEGER PRIMARY KEY,MBthred_id INTEGER NOT NULL,FOREIGN KEY (MBthred_id) REFERENCES messageBoard(MBthred_id),user_id INTEGER NOT NULL,FOREIGN KEY(user_id) REFERENCES accountHolder(user_id), message TEXT NOT NULL, MessageTimestamp DATE NOT NULL);

CREATE TABLE DirectMessages(DM_ID INTEGER PRIMARY KEY, Head_User INTEGER NOT NULL, FOREIGN KEY (Head_User) REFERENCES accountHolder(user_id), Other_User INTEGER NOT NULL, FOREIGN KEY (Other_User) REFERENCES accountHolder(user_id), MessageTimestamp DATE, Message TEXT NOT NULL)
